A placeholder loading component that shows a pulsing animation while content is loading. Use skeletons to reduce perceived loading time and prevent layout shift.

## Installation

<RegistrySetup />

<Tabs items={['CLI', 'Manual']}>
  <Tab value="CLI">
    ```bash
    npx shadcn@latest add @nteract/skeleton
    ```
  </Tab>
  <Tab value="Manual">
    Copy the component from this repository's `registry/primitives/skeleton.tsx`.
  </Tab>
</Tabs>

## Usage

```tsx
import { Skeleton } from "@/registry/primitives/skeleton"

export function Example() {
  return <Skeleton className="h-4 w-[200px]" />
}
```
