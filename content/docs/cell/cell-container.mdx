A wrapper component for notebook cells that handles:
- Focus/selection border styling
- `data-cell-id` attribute for cell identification
- Optional drag-and-drop event handlers

## Usage

```tsx
import { CellContainer } from "@/registry/cell/CellContainer"

export function NotebookCell({ cellId, isFocused, onFocus }) {
  return (
    <CellContainer
      id={cellId}
      isFocused={isFocused}
      onFocus={onFocus}
    >
      {/* Cell content */}
    </CellContainer>
  )
}
```

```tsx

  id="cell-1"
  isFocused={false}
  onDragStart={(e) => {
    e.dataTransfer.setData("text/plain", "cell-1");
  }}
  onDragOver={(e) => {
    e.preventDefault();
    e.dataTransfer.dropEffect = "move";
  }}
  onDrop={(e) => {
    e.preventDefault();
    const draggedId = e.dataTransfer.getData("text/plain");

  }}

  <div className="p-4">Draggable cell</div>
```

## Props

| Prop | Type | Default | Description |
| --- | --- | --- | --- |
| `id` | `string` | — | Cell identifier, used for `data-cell-id` attribute |
| `isFocused` | `boolean` | `false` | Whether the cell is currently focused |
| `onFocus` | `() => void` | — | Callback when cell is clicked |
| `children` | `ReactNode` | — | Cell content |
| `onDragStart` | `(e: React.DragEvent) => void` | — | Drag start handler (enables draggable) |
| `onDragOver` | `(e: React.DragEvent) => void` | — | Drag over handler |
| `onDrop` | `(e: React.DragEvent) => void` | — | Drop handler |
| `className` | `string` | — | Additional CSS classes |
| `focusBgColor` | `string` | `"bg-primary/5"` | Background color when focused |
| `focusBorderColor` | `string` | `"border-primary/60"` | Border color when focused |
